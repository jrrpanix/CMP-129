   public boolean equals( Object o ) {
      if ( o.getClass() != Rectangle.class ) return false;
      return equals( (Rectangle)o );  // <- Casting Object to Rectangle
   }

   public boolean equals( Rectangle r ) {
      return len_ == r.len_ && width_ == r.width_;
   }
