In this lecture we are going to make a simple user interface and then
add some Event Listeners to allow us to make front end more dynamic
in behavior

   (*) We will cover "ActionListener" and "KeyListener"

   (*) "ActionListener" 
       interface for when a JButton  is pressed or the key "Enter" is hit
       for a JTextField

   (*) "KeyListener" is used when a Key is Pressed, Key is Type or Released.
       certian keys are Pressed but not typed, examples include "Ctrl", "Shift"
       "Alt" 

(*) ActionListener
    
    this interfacde has the method 

        void actionPerformed( ActionEvent e )

(*) KeyPressed 

    this interface has the methods 

        void keyPressed( KeyEvent e );
        void keyTyped( KeyEvent e );
        void keyReleased( KeyEvent e );


(*) We are going to make up a Simple User Interface that contains a JLable
    and a JTextField




import javax.swing.*;
import java.awt.*;
import java.awt.event.*;

public class StringEntryView extends JFrame{

       JPanel     panel;
       JTextField text;
       JLabel     label;
       
       public StringEntryView() {
         super("String Entry Example");
	 initFrame();
	 Build();
	 setVisible(true);
      }
					
     private void build() {
	panel = new JPanel();
        label  = new JLabel( "Word:");
    	text = new JTextField(15);
        text.addActionListener(new TextListener()); <--- calls our implementation of ActionListener
        text.addKeyListener(new TextKeyListener()); <--- calls our implementation of KeyListener
  	panel.add(label);
        panel.add(text);
        add(panel);
     }

     private void initFrame() {
       int width = 275, height= 75;
       setSize(width,height);
       setLocation(500,500);
       setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
    }

    //
    // This class implements ActionListener
    // We have to implement the method "actionPerformed( ActionEvent e)"
    //
    class TextListener implements ActionListener{

 	 public void actionPerformed( ActionEvent e) {    
             JOptionPane.showMessageDialog(text,new String("Text:" +text.getText()));
	 }
     }

    //
    // This class implements KeyListener
    // We have to implement the methods 
    //
    //        public void keyPressed( KeyEvent e)
    //        public void keyTyped( KeyEvent e)
    //        public void keyReleased( KeyEvent e)
    //
    class TextKeyListener implements KeyListener{

        public void keyPressed( KeyEvent e){
            summarizeAction(e,"Pressed");
        }

        public void keyTyped( KeyEvent e) {
            summarizeAction(e,"Typed");
        }

        public void keyReleased( KeyEvent e) {
     	    summarizeAction(e,"Released");
        }

        public void summarizeAction( KeyEvent e, String a){
           System.out.println( "you +" + a + " " + e.getKeyChar());
        }
    }

    public static void main(String [] args ){
        STringEntryView view = new StringEntryView();
    }

}
